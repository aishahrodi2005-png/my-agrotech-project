    <!-- Interactive Logic Script -->
    <script>
        // Constants mapping directly to Malaysian Chilli (Kulai) growth trials
        const L = 7800;   // Carrying capacity / maximum leaf area in cm^2
        const k = 0.085;  // Intrinsic growth factor scaling per day
        const x0 = 56;    // Midpoint trajectory node (inflection point at Day 56)

        // Generating core analytical data streams
        const totalDays = 112;
        const labels = [];
        const areaData = [];
        const rateData = [];

        for (let day = 0; day <= totalDays; day += 2) {
            labels.push(day);
            
            // Core Logistic Equation Evaluation
            const exponentialTerm = Math.exp(-k * (day - x0));
            const calculatedArea = L / (1 + exponentialTerm);
            areaData.push(calculatedArea.toFixed(1));

            // Instantaneous Derivative Evaluation
            const derivativeRate = (L * k * exponentialTerm) / Math.pow(1 + exponentialTerm, 2);
            rateData.push(derivativeRate.toFixed(2));
        }

        // Initialize Chart.js configuration context
        const ctx = document.getElementById('growthChart').getContext('2d');
        const growthChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [
                    {
                        label: 'Leaf Area f(x) (cm²)',
                        data: areaData,
                        borderColor: '#059669',
                        backgroundColor: 'rgba(5, 150, 105, 0.04)',
                        borderWidth: 3,
                        yAxisID: 'yArea',
                        tension: 0.2,
                        pointRadius: 2,
                        pointHoverRadius: 6,
                        fill: true
                    },
                    {
                        label: "Growth Velocity f'(x) (cm²/day)",
                        data: rateData,
                        borderColor: '#3b82f6',
                        borderDash: [5, 5],
                        borderWidth: 2,
                        yAxisID: 'yRate',
                        tension: 0.2,
                        pointRadius: 0,
                        pointHoverRadius: 6,
                        fill: false
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                interaction: {
                    mode: 'index',
                    intersect: false,
                },
                scales: {
                    x: {
                        title: {
                            display: true,
                            text: 'Days After Transplanting (DAT)',
                            color: '#475569',
                            font: { weight: '600', size: 11 }
                        },
                        grid: { color: '#f1f5f9' }
                    },
                    yArea: {
                        type: 'linear',
                        position: 'left',
                        title: {
                            display: true,
                            text: 'Cumulative Leaf Surface Area (cm²)',
                            color: '#059669',
                            font: { weight: '600', size: 11 }
                        },
                        min: 0,
                        max: 8500,
                        grid: { color: '#f1f5f9' }
                    },
                    yRate: {
                        type: 'linear',
                        position: 'right',
                        title: {
                            display: true,
                            text: 'Instant Expansion Rate (cm²/day)',
                            color: '#3b82f6',
                            font: { weight: '600', size: 11 }
                        },
                        min: 0,
                        max: 200,
                        grid: { drawOnChartArea: false } // Avoid secondary grid clutter
                    }
                },
                onClick: (event, activeElements) => {
                    if (activeElements.length > 0) {
                        const pointIndex = activeElements[0].index;
                        const day = labels[pointIndex];
                        const area = areaData[pointIndex];
                        const derivative = rateData[pointIndex];
                        
                        renderTelemetryPayload(day, area, derivative);
                    }
                }
            }
        });

        // Telemetry calculation formatter engine
        function renderTelemetryPayload(day, area, derivative) {
            document.getElementById('promptInstructions').classList.add('hidden');
            const dataLayout = document.getElementById('calculationPayload');
            dataLayout.classList.remove('hidden');

            document.getElementById('readoutDay').innerText = `Day ${day}`;
            document.getElementById('readoutArea').innerText = `${parseFloat(area).toLocaleString()} cm²`;
            document.getElementById('readoutDerivative').innerText = `${derivative} cm²/day`;

            let laymanExplanation = "";

            if (day < 30) {
                laymanExplanation = `Early seedling establishment stage. The derivative is relatively small (${derivative} cm²/day) because the plant has limited foliage to intercept sunlight. Focus on root development solutions in your fertigation setup right now.`;
            } else if (day >= 30 && day < 52) {
                laymanExplanation = `Vegetative explosion phase! The derivative is accelerating rapidly. The plant is adding more leaf tissue every single day compared to the day before. Nitrogen demands are peaking during this window to construct proteins.`;
            } else if (day >= 52 && day <= 60) {
                laymanExplanation = `Inflection point reached around Day 56! The growth velocity has topped out at its absolute maximum (${derivative} cm²/day). This derivative peak marks the shift: the rate of leaf expansion will now begin to slow as the crop changes priority from vegetative tissue to flowering blocks.`;
            } else if (day > 60 && day < 90) {
                laymanExplanation = `Canopy maturation phase. Although total leaf area continues to climb up to ${parseFloat(area).toLocaleString()} cm², the derivative is dropping. Vegetative expansion is slowing because the plant is redirecting sugars and minerals to support the growth of early chilli pods.`;
            } else {
                laymanExplanation = `Full canopy stabilization. The derivative is practically flat near zero (${derivative} cm²/day). Leaf generation has balanced out with pruning and drop. Your canopy structure is locked in; any additional nutrients are pumped straight into ripening Chilli Kulai yields for market harvest.`;
            }

            document.getElementById('readoutExplanation').innerHTML = laymanExplanation;
        }
    </script>
